#include <iostream>
#include <list>
#include <vector>
#include <algorithm>
using namespace std;


int main()
{
	vector<list<int>> vec(7);     //  创建7个链表
	for(int i=0;i<7;i++)
	{
		for(int j=0;j<=i;j++)
		{
			vec[i].push_back(j+1);     // vec[i] ===> lst
		}
	}

	//---------------------------------------------------------
	for(int i=0;i<7;i++)
	{
		list<int>::iterator ite = vec[i].begin();
		while(ite != vec[i].end())
		{
			cout << *ite << " ";
			++ite;
		}
		cout << endl;
	}



	//list<vector<int>> lst;
	//list<vector<int>>::iterator ite;
	//(*ite)[0];



	system("pause");
	return 0;
}
